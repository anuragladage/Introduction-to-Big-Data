SELECT a_row as row_num,b_col as col_num, SUM(value) as value
FROM 
	(SELECT a.row_num as a_row,b.col_num as b_col,a.value*b.value as value 
	FROM a,b 
	WHERE a.col_num==b.row_num)
GROUP BY a_row,b_col;
