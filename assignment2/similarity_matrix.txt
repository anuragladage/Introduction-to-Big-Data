CREATE TABLE frequency_T AS
SELECT term as docid_t,docid as term_t,count 
FROM frequency;


SELECT * 
FROM
	(SELECT docid as row_name,docid_t as col_name, SUM(count) as SIMILARITY_SCORE
	FROM 
		(SELECT docid as docid,docid_t as docid_t,frequency.count*frequency_T.count as count 
		FROM frequency,frequency_T 
		WHERE frequency.term==frequency_T.term_t)
	GROUP BY docid,docid_t)
WHERE row_name = '10080_txt_crude' AND col_name = '17035_txt_earn';
